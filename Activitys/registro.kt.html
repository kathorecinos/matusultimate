1    package Activitys
2    
3    import android.content.Intent
4    import android.os.Bundle
5    import android.widget.Toast
6    import androidx.appcompat.app.AppCompatActivity

8    import androidx.core.content.ContextCompat
9    import com.example.matusintl.databinding.ActivityRegistroBinding
10   import com.google.firebase.auth.FirebaseAuth
11   import java.util.concurrent.Executor
12   
13   class registro : AppCompatActivity() {
14   
15       private lateinit var binding: ActivityRegistroBinding
16       private lateinit var executor: Executor
17       private lateinit var biometricPrompt: BiometricPrompt
18       private lateinit var promptInfo: BiometricPrompt.PromptInfo
19   
20       private lateinit var auth: FirebaseAuth
21   
22       override fun onCreate(savedInstanceState: Bundle?) {
23           super.onCreate(savedInstanceState)
24           binding = ActivityRegistroBinding.inflate(layoutInflater)
25           val view = binding.root
26           setContentView(view)
27           supportActionBar?.hide()
28   
29           executor = ContextCompat.getMainExecutor(this)
30   
31           biometricPrompt = BiometricPrompt(
32               this@registro,
33               executor,
34               object : BiometricPrompt.AuthenticationCallback() {
35                   override fun onAuthenticationError(errorCode: Int, errString: CharSequence) {
36                       super.onAuthenticationError(errorCode, errString)
37                       Toast.makeText(this@registro, "Vuelva a ingresar su huella", Toast.LENGTH_SHORT)
38                           .show()
39                   }
40   
41                   override fun onAuthenticationSucceeded(result: BiometricPrompt.AuthenticationResult) {
42                       super.onAuthenticationSucceeded(result)
43                       Toast.makeText(this@registro, "Acceso Aceptado", Toast.LENGTH_SHORT).show()
44                       startActivity(Intent(this@registro, main::class.java))
45   
46                   }
47   
48                   override fun onAuthenticationFailed() {
49                       super.onAuthenticationFailed()
50                       Toast.makeText(this@registro, "Vuelva a ingresar su huella", Toast.LENGTH_SHORT)
51                           .show()
52                   }
53   
54               })
55           promptInfo= BiometricPrompt.PromptInfo.Builder()
56               .setTitle("Ingrese su huella")
57               .setSubtitle("Para ingresar a su MatusApp solo ponga su dedo en la huella para reconocimiento dactilar")
58               .setNegativeButtonText("Usar contrase√±a")
59               .build()
60   
61           binding.button2.setOnClickListener {
62               biometricPrompt.authenticate(promptInfo)
63           }
64   
65   
66           auth = FirebaseAuth.getInstance()
67   
68           binding.button1.setOnClickListener {
69              if(binding.mail.text?.trim().toString().isNotEmpty()||binding.password.text?.trim().toString().isNotEmpty()) {
70                  signInUser(
71                      binding.mail.text?.trim().toString(),
72                      binding.password.text?.trim().toString()
73                  )
74              }else{
75                  Toast.makeText(this, "Input required", Toast.LENGTH_LONG).show()
76              }
77           }
78       }
79       
80   
81       fun signInUser(mail:String,password:String){
82   
83           auth.signInWithEmailAndPassword(mail,password)
84               .addOnCompleteListener (this) {task ->
85   
86                   if (task.isSuccessful){
87                       val intent = Intent(this, main::class.java)
88                       startActivity(intent)
89                   }else{
90                       Toast.makeText(this, "Error!!"+task.exception, Toast.LENGTH_LONG).show()
91                   }
92               }
93   
94   
95   
96       }
97   }
