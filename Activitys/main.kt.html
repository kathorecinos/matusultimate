1    package Activitys
2    
3    import android.content.Intent
4    import android.os.Bundle
5    import android.view.View
6    import androidx.appcompat.app.AppCompatActivity
7    import androidx.fragment.app.Fragment
8    import com.example.matusintl.R
9    import com.example.matusintl.databinding.ActivityMainBinding
10   import com.google.firebase.auth.FirebaseAuth
11   
12   
13   class main : AppCompatActivity() {
14   
15       private lateinit var binding: ActivityMainBinding
16       private val userFragment = UserFragment()
17       private val trackerFragment = trackerFragment()
18       private val historyFragment = historyFragment()
19       private val galleryFragment = galleryFragment()
20       private val dashboardFragment = dashboardFragment()
21       private lateinit var mAuth: FirebaseAuth
22   
23   
24       override fun onCreate(savedInstanceState: Bundle?) {
25           super.onCreate(savedInstanceState)
26           binding = ActivityMainBinding.inflate(layoutInflater)
27           val view = binding.root
28           setContentView(view)
29           replaceFragment(dashboardFragment)
30           mAuth = FirebaseAuth.getInstance()
31   
32   
33           binding.bottomNav.setOnNavigationItemSelectedListener {
34               when (it.itemId) {
35                   R.id.inicio -> replaceFragment(dashboardFragment)
36                   R.id.Tracking -> replaceFragment(trackerFragment)
37                   R.id.history -> replaceFragment(historyFragment)
38                   R.id.gallery -> replaceFragment(galleryFragment)
39                   R.id.usuario -> replaceFragment(userFragment)
40               }
41               true
42           }
43       }
44   
45       fun onClick(view: View) {
46           when (view.id) {
47               R.id.out -> {
48                   mAuth.signOut()
49                   startActivity(Intent(this, Out::class.java))
50                   finish()
51               }
52           }
53       }
54   
55       private fun replaceFragment(fragment: Fragment) {
56           val transaction = supportFragmentManager.beginTransaction()
57           transaction.replace(R.id.fragmentContainer, fragment)
58           transaction.commit()
59   
60       }
61   }
