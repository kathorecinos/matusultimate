77       private fun uploadFile() {
78           if (filepath!=null){
79               var pd=ProgressDialog(this)
80               pd.setTitle("Uploading")
81               pd.show()
82   
83               var imageRef = FirebaseStorage.getInstance().reference.child("images/pic.jpg")
84               imageRef.putFile(filepath)
85                   .addOnSuccessListener {p0 ->
86                       pd.dismiss()
87                       Toast.makeText(applicationContext,"File Uploaded", Toast.LENGTH_LONG).show()
88                   }
89                   .addOnFailureListener {p0 ->
90                       pd.dismiss()
91                       Toast.makeText(applicationContext, p0.message, Toast.LENGTH_LONG).show()
92   
93                   }
94                   .addOnProgressListener { p0 ->
95                       var progress = (100.0 * p0.bytesTransferred) / p0.totalByteCount
96                       pd.setMessage("Uploaded ${progress.toInt()}%")
97   
98                   }
99           }
100      }

109      override fun onActivityResult(requestCode: Int, resultCode: Int, data: Intent?) {
110          super.onActivityResult(requestCode, resultCode, data)

112              filepath = data.data!!

119      private fun saveFireStore(
120          Nombre: String,
121          Importadora: String,
122          Correo: String,
123          Telefono: String
124      ) {
125          val db = FirebaseFirestore.getInstance()
126          val user: MutableMap<String, Any> = HashMap()
127          user["Nombre"] = Nombre
128          user["Importadora"] = Importadora
129          user["Correo"] = Correo
130          user["Telefono"] = Telefono
131  
132          db.collection("users")
133              .add(user)
134              .addOnSuccessListener {
135                  Toast.makeText(this@userData, "record added succesfully", Toast.LENGTH_LONG).show()
136                  var intent = Intent(this, dataSendActivity::class.java)
137                  startActivity(intent)
138              }
139              .addOnFailureListener {
140                  Toast.makeText(this@userData,"record failed", Toast.LENGTH_LONG).show()
141              }
